// نظام الدعم الفني الذكي اللي سويناه لمشروع وجهة

                    
// تم التطوير بأيدينا: يوسف المطيري

                    
 

                    
class AISupportBot {

                    
  constructor() {

                    
    this.context = "دعم فني متخصص في تأجير السيارات";

                    
    this.expertise = ["الحجوزات", "المدفوعات", "الصيانة", "التأمين"];

                    
  }

                    
 

                    
  async analyzeQuery(userQuery) {

                    
    // أحلل استفسارك باستخدام معالجة اللغة العربية

                    
    const intent = await this.detectIntent(userQuery);

                    
    const context = this.addBusinessContext(intent);

                    
    return this.generateResponse(context);

                    
  }

                    
}

بداية الطموح في جامعة حفر الباطن

                    
دخلت تخصص نظم المعلومات الإدارية (MIS) في كلية إدارة الأعمال، وبدأت أفهم إني ما أبغى أكون مبرمج بس، أبغى أفهم البيزنس وأطبق التقنية عليه.

                

                
                

                    
2025

                    
ولادة منصة "وجهة"

                    
بعد ما تعلمت وشفت الفرص، سويت أول نظام متكامل لتأجير السيارات من الفكرة للواقع. وحدة من أصعب وأجمل التحديات اللي واجهتها.

                

                
                

                    
2026

                    
التخرج والإنجاز

                    
وقفنا على أعتاب سوق العمل، أحمل معي شهادة MIS وخبرة عملية حقيقية. جاهز أغير طريقة عمل الشركات وأسوي أثر حقيقي.
